Banco Cine totem - Modelo

CREATE TABLE cine.usuarios(
	pkcod_usuario INTEGER AUTO_INCREMENT primary key,
	nome VARCHAR(200),
	login VARCHAR(200),	
	senha VARCHAR(200),
	cpf VARCHAR(200),
	saldo REAL,	
	
);

CREATE TABLE cine.filmes (
	pkcod_filme INTEGER AUTO_INCREMENT primary key,
	nome VARCHAR(200),
	sinopse VARCHAR(200),
	valor double,
	duracao double,
	classificacao INTEGER,
	
);

CREATE TABLE cine.salas (
	pkcod_sala INTEGER AUTO_INCREMENT,
	fkcod_filme INTEGER,	
	numero_sala INTEGER,
	FOREIGN KEY (fkcod_filme) REFERENCES Filmes(pkcod_filme),
	PRIMARY KEY(pkcod_sala)
);
