Banco Cine totem - Modelo

CREATE TABLE Usuarios(
	pkcod_usuario INTEGER AUTO_INCREMENT,
	nome VARCHAR(200),
	login VARCHAR(200),	
	senha VARCHAR(200),
	cpf VARCHAR(200),
	saldo REAL,	
	PRIMARY KEY(pkcod_usuario)
);

CREATE TABLE Filmes (
	pkcod_filme INTEGER AUTO_INCREMENT,
	nome VARCHAR(200),
	descricao VARCHAR(200),
	valor REAL,
	duracao REAL,
	classificacao INTEGER,
	imagem VARCHAR(200),
	PRIMARY KEY(pkcod_filme)
);

CREATE TABLE Salas (
	pkcod_sala INTEGER AUTO_INCREMENT,
	fkcod_filme INTEGER,	
	numero_sala INTEGER,
	FOREIGN KEY (fkcod_filme) REFERENCES Filmes(pkcod_filme),
	PRIMARY KEY(pkcod_sala)
);
